Hints for installation on Windows

After installing git-bash add

#Make zsh default shell
if [ -t 1 ]; then
  exec zsh
fi

to bash.bashrc

To .zshrc

export PYTHONIOENCODING=UTF-8

 | sed -e 's/"$CONDA_EXE" $_CE_M $_CE_CONDA "$@"/"$CONDA_EXE" $_CE_M $_CE_CONDA "$@" | tr -d \x27\\r\x27/g'
